#include <stdio.h>
#include <string.h>
void countThings(char *fileName, int *wordCount, int *charCount)
{
FILE *fp = fopen(fileName, "r");
char string[100];
while(1)
{
fscanf(fp, "%s", string);
//printf("%s\n", string);
(*wordCount)++;
*charCount += strlen(string);
if(feof(fp))
break;
}
}
int main()
{
char fileName[] = "Story.txt";
int wordCount = 0, charCount = 0;
countThings(fileName, &wordCount, &charCount);
printf("Words: %d\nCharacters (no spaces): %d\n", wordCount, charCount);
}
